import { BlendedNoise, ImprovedNoise, NormalNoise, PerlinNoise, PerlinSimplexNoise, Random, SimplexNoise } from 'deepslate'
import Plot from '@site/src/components/Plot'

export const SIZE = 240
export const SEED = BigInt(6203)
export const gray = (min, max) => value => (value - min) / (max - min) * 256

## ImprovedNoise

```ts
const noise = new ImprovedNoise(new Random(BigInt(6203)))
const value = noise.sample(x / 32, y / 32, 0)
```

<Plot name="Improved" width={SIZE} colorizer={gray(-1, 1)}
	initializer={() => new ImprovedNoise(new Random(SEED))}
	sampler={noise => (x, y) => noise.sample(x / 32, y / 32, 0)}
/>

## BlendedNoise

```ts

const noise = new BlendedNoise(new Random(BigInt(6203)), sampling)
const value = noise.sample(x, y, 0)
```
